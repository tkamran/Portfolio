#define a function that returns a list of indices of the outliers
def outliers(df, ft):
    Q1 = df[ft].quantile(0.25)
    Q3 = df[ft].quantile(0.75)
    IQR = Q3 - Q1
    
    lower = Q1 - 1.5 * IQR
    upper = Q3 + 1.5 * IQR

    
    outliers = df.index[ (df[ft] < lower) | (df[ft] > upper)]
    
    return outliers

def timezone(x):
    if x>=datetime.time(5, 0, 1) and x <=datetime.time(12, 0, 0):
        return 'morning'
    elif x>=datetime.time(12, 0, 1) and x <=datetime.time(17, 0, 0):
        return 'midday'
    elif x>=datetime.time(17, 0, 1) and x <=datetime.time(21, 0, 0):
        return 'evening'
    elif x>=datetime.time(21, 0, 1) or x <=datetime.time(5, 0, 0):
        return 'late night'
